// Custom Web Audio API Emotional Music Synthesizer
class ApologySynth {
  private ctx: AudioContext | null = null;
  private pianoGain: GainNode | null = null;
  private violinGain: GainNode | null = null;
  private masterGain: GainNode | null = null;
  private timerId: any = null;
  private chordIndex = 0;
  private isPlaying = false;

  private chords = [
    [220, 261.63, 329.63, 440], // Am (A3, C4, E4, A4)
    [174.61, 220, 261.63, 349.23], // F (F3, A3, C4, F4)
    [261.63, 329.63, 392, 523.25], // C (C3, E3, G3, C4)
    [196, 246.94, 293.66, 392] // G (G3, B3, D4, G4)
  ];
  private violinMelody = [440, 493.88, 523.25, 587.33, 659.25, 587.33, 523.25, 493.88]; // A4, B4, C5, D5, E5...

  public start() {
    if (this.isPlaying) return;
    try {
      this.ctx = new (window.AudioContext || (window as any).webkitAudioContext)();
      
      this.masterGain = this.ctx.createGain();
      this.masterGain.gain.setValueAtTime(0.35, this.ctx.currentTime);
      this.masterGain.connect(this.ctx.destination);

      this.pianoGain = this.ctx.createGain();
      this.pianoGain.gain.setValueAtTime(0.5, this.ctx.currentTime);
      this.pianoGain.connect(this.masterGain);

      this.violinGain = this.ctx.createGain();
      this.violinGain.gain.setValueAtTime(0.2, this.ctx.currentTime);
      this.violinGain.connect(this.masterGain);

      this.isPlaying = true;
      this.chordIndex = 0;
      this.scheduler();
    } catch (e) {
      console.error("Audio Context failed to start:", e);
    }
  }

  private scheduler() {
    if (!this.isPlaying || !this.ctx) return;

    const playStep = () => {
      this.playPianoChord(this.chords[this.chordIndex % this.chords.length]);
      if (this.chordIndex % 2 === 0) {
        this.playViolinNote(this.violinMelody[Math.floor(this.chordIndex / 2) % this.violinMelody.length]);
      }
      this.chordIndex++;
      this.timerId = setTimeout(playStep, 4000);
    };

    playStep();
  }

  private playPianoChord(frequencies: number[]) {
    if (!this.ctx || !this.pianoGain) return;
    const now = this.ctx.currentTime;

    frequencies.forEach((freq, idx) => {
      const osc = this.ctx!.createOscillator();
      const gain = this.ctx!.createGain();
      const filter = this.ctx!.createBiquadFilter();

      // Arpeggiate chord notes slightly for emotional, delicate piano feel
      const noteDelay = idx * 0.08;

      osc.type = "triangle";
      osc.frequency.setValueAtTime(freq, now + noteDelay);

      filter.type = "lowpass";
      filter.frequency.setValueAtTime(800, now);

      gain.gain.setValueAtTime(0, now + noteDelay);
      gain.gain.linearRampToValueAtTime(0.12, now + noteDelay + 0.04);
      gain.gain.exponentialRampToValueAtTime(0.001, now + noteDelay + 3.5);

      osc.connect(filter);
      filter.connect(gain);
      gain.connect(this.pianoGain!);

      osc.start(now + noteDelay);
      osc.stop(now + noteDelay + 3.6);
    });
  }

  private playViolinNote(frequency: number) {
    if (!this.ctx || !this.violinGain) return;
    const now = this.ctx.currentTime;

    const osc = this.ctx.createOscillator();
    const vibrato = this.ctx.createOscillator();
    const vibratoGain = this.ctx.createGain();
    const gain = this.ctx.createGain();
    const filter = this.ctx.createBiquadFilter();

    osc.type = "sine";
    // Soaring strings play higher registers
    osc.frequency.setValueAtTime(frequency * 1.5, now);

    vibrato.frequency.value = 4.8; 
    vibratoGain.gain.value = 6; 

    vibrato.connect(vibratoGain);
    vibratoGain.connect(osc.frequency);

    filter.type = "lowpass";
    filter.frequency.setValueAtTime(600, now);
    filter.frequency.linearRampToValueAtTime(1200, now + 3);

    gain.gain.setValueAtTime(0, now);
    gain.gain.linearRampToValueAtTime(0.06, now + 1.2);
    gain.gain.setValueAtTime(0.06, now + 5.0);
    gain.gain.exponentialRampToValueAtTime(0.001, now + 7.5);

    osc.connect(filter);
    filter.connect(gain);
    gain.connect(this.violinGain);

    vibrato.start(now);
    osc.start(now);
    osc.stop(now + 7.6);
    vibrato.stop(now + 7.6);
  }

  public stop() {
    this.isPlaying = false;
    if (this.timerId) clearTimeout(this.timerId);
    if (this.ctx) {
      try {
        this.ctx.close();
      } catch (e) {
        console.error(e);
      }
      this.ctx = null;
    }
  }

  public toggleMute(muted: boolean) {
    if (this.masterGain && this.ctx) {
      this.masterGain.gain.setValueAtTime(muted ? 0 : 0.35, this.ctx.currentTime);
    }
  }
}

// Global synth instance
const audioSynth = new ApologySynth();

export default function App() {
  const [isPlaying, setIsPlaying] = useState(false);
  const [isAccepted, setIsAccepted] = useState(false);
  const [showHeartsRain, setShowHeartsRain] = useState(false);
  const canvasRef = useRef<HTMLCanvasElement | null>(null);
  const mousePos = useRef({ x: 0, y: 0 });

  // Handle music state
  const handleMusicToggle = () => {
    if (!isPlaying) {
      audioSynth.start();
      setIsPlaying(true);
    } else {
      audioSynth.stop();
      setIsPlaying(false);
    }
  };

  // Scroll function
  const scrollToId = (id: string) => {
    const el = document.getElementById(id);
    if (el) {
      el.scrollIntoView({ behavior: "smooth" });
    }
  };

  // Trigger Accept Ceremony
  const handleAcceptApology = () => {
    setIsAccepted(true);
    setShowHeartsRain(true);
    // Spawns mass heart rain triggers in canvas
    setTimeout(() => {
      setShowHeartsRain(false);
    }, 6000);
  };

  // Canvas particle effects
  useEffect(() => {
    const canvas = canvasRef.current;
    if (!canvas) return;
    const ctx = canvas.getContext("2d");
    if (!ctx) return;

    let animationFrameId: number;
    let particles: Array<any> = [];

    const resizeCanvas = () => {
      canvas.width = window.innerWidth;
      canvas.height = window.innerHeight;
    };

    resizeCanvas();
    window.addEventListener("resize", resizeCanvas);

    // Particle classes
    class Sparkle {
      x: number;
      y: number;
      vx: number;
      vy: number;
      size: number;
      color: string;
      alpha: number;
      decay: number;

      constructor(x: number, y: number, isExplosion = false) {
        this.x = x;
        this.y = y;
        const angle = Math.random() * Math.PI * 2;
        const speed = isExplosion ? Math.random() * 5 + 2 : Math.random() * 1.5 + 0.5;
        this.vx = Math.cos(angle) * speed;
        this.vy = Math.sin(angle) * speed - (isExplosion ? 1 : 0.5); // drift up
        this.size = Math.random() * 3 + 1.5;
        this.color = `hsl(${40 + Math.random() * 15}, 100%, ${65 + Math.random() * 20}%)`; // luxury gold colors
        this.alpha = 1;
        this.decay = Math.random() * 0.015 + 0.01;
      }

      update() {
        this.x += this.vx;
        this.y += this.vy;
        this.alpha -= this.decay;
      }

      draw(c: CanvasRenderingContext2D) {
        c.save();
        c.globalAlpha = this.alpha;
        c.shadowBlur = 8;
        c.shadowColor = this.color;
        c.fillStyle = this.color;
        c.beginPath();
        c.arc(this.x, this.y, this.size, 0, Math.PI * 2);
        c.fill();
        c.restore();
      }
    }

    class FloatingHeart {
      x: number;
      y: number;
      size: number;
      vy: number;
      vx: number;
      alpha: number;
      scale: number;
      growth: number;
      color: string;
      isExplosion: boolean;

      constructor(x: number, y: number, isExplosion = false, direction = -1) {
        this.x = x;
        this.y = y;
        this.isExplosion = isExplosion;
        this.size = Math.random() * 12 + 8;
        if (isExplosion) {
          const angle = Math.random() * Math.PI * 2;
          const speed = Math.random() * 6 + 3;
          this.vx = Math.cos(angle) * speed;
          this.vy = Math.sin(angle) * speed;
          this.alpha = 1;
        } else {
          this.vx = (Math.random() - 0.5) * 0.8;
          this.vy = (Math.random() * 1 + 0.8) * direction; // can fall or rise
          this.alpha = Math.random() * 0.5 + 0.4;
        }
        this.scale = 1;
        this.growth = Math.random() * 0.02 - 0.01;
        this.color = `rgba(${255}, ${Math.floor(Math.random() * 50 + 40)}, ${Math.floor(Math.random() * 80 + 90)}, `;
      }

      update() {
        this.x += this.vx;
        this.y += this.vy;
        if (this.isExplosion) {
          this.vy += 0.05; // gravity for explosion
          this.alpha -= 0.018;
        } else {
          this.alpha -= 0.0015;
        }
        this.scale += this.growth;
      }

      draw(c: CanvasRenderingContext2D) {
        if (this.alpha <= 0) return;
        c.save();
        c.globalAlpha = this.alpha;
        c.fillStyle = this.isExplosion ? "#FF4D6D" : "rgba(255, 77, 109, 0.7)";
        c.shadowBlur = this.isExplosion ? 12 : 4;
        c.shadowColor = "#FF4D6D";
        
        c.translate(this.x, this.y);
        c.scale(this.scale * 0.7, this.scale * 0.7);
        c.beginPath();
        // Standard heart shape curve
        c.moveTo(0, 0);
        c.bezierCurveTo(-this.size / 2, -this.size / 2, -this.size, this.size / 3, 0, this.size);
        c.bezierCurveTo(this.size, this.size / 3, this.size / 2, -this.size / 2, 0, 0);
        c.closePath();
        c.fill();
        c.restore();
      }
    }

    class GoldenButterfly {
      x: number;
      y: number;
      vx: number;
      vy: number;
      angle: number;
      flapSpeed: number;
      size: number;
      color: string;
      alpha: number;

      constructor() {
        this.x = Math.random() * window.innerWidth;
        this.y = Math.random() * window.innerHeight;
        this.vx = (Math.random() - 0.5) * 1.5;
        this.vy = (Math.random() - 0.6) * 1.2;
        this.angle = Math.random() * Math.PI * 2;
        this.flapSpeed = 0.08 + Math.random() * 0.05;
        this.size = Math.random() * 10 + 10;
        this.color = "#FFD700";
        this.alpha = Math.random() * 0.6 + 0.4;
      }

      update() {
        this.x += this.vx;
        this.y += this.vy;
        this.angle += this.flapSpeed;

        // Periodic direction change
        if (Math.random() < 0.02) {
          this.vx = (Math.random() - 0.5) * 1.5;
          this.vy = (Math.random() - 0.6) * 1.2;
        }

        // Keep inside bounds gracefully
        if (this.x < -20) this.x = window.innerWidth + 20;
        if (this.x > window.innerWidth + 20) this.x = -20;
        if (this.y < -20) this.y = window.innerHeight + 20;
        if (this.y > window.innerHeight + 20) this.y = -20;
      }

      draw(c: CanvasRenderingContext2D) {
        c.save();
        c.globalAlpha = this.alpha;
        c.shadowBlur = 10;
        c.shadowColor = this.color;
        c.fillStyle = this.color;
        c.translate(this.x, this.y);

        // Draw animated butterfly wings
        const flap = Math.abs(Math.sin(this.angle));
        c.beginPath();
        // Left Wing
        c.ellipse(-this.size/2 * flap, -this.size/4, this.size/2 * flap, this.size/2, -Math.PI/6, 0, Math.PI*2);
        // Right Wing
        c.ellipse(this.size/2 * flap, -this.size/4, this.size/2 * flap, this.size/2, Math.PI/6, 0, Math.PI*2);
        c.fill();

        // Little antenna
        c.strokeStyle = "rgba(255,215,0,0.8)";
        c.lineWidth = 1;
        c.beginPath();
        c.moveTo(0, -this.size/2);
        c.quadraticCurveTo(-3, -this.size - 2, -6, -this.size);
        c.moveTo(0, -this.size/2);
        c.quadraticCurveTo(3, -this.size - 2, 6, -this.size);
        c.stroke();

        c.restore();
      }
    }

      y: number;
      emoji: string;
      vy: number;
      vx: number;
      alpha: number;
      size: number;

      constructor(x: number, emoji: string) {

        this.alpha = Math.random() * 0.5 + 0.3;
        this.size = Math.random() * 16 + 24;
      }

      update() {
        this.x += this.vx;
        this.y += this.vy;
        if (this.y < -50) {
          this.y = window.innerHeight + 50;
          this.alpha = Math.random() * 0.5 + 0.3;
        }
      }

      draw(c: CanvasRenderingContext2D) {
        c.save();
        c.globalAlpha = this.alpha;
        c.font = `${this.size}px sans-serif`;
        c.fillText(this.emoji, this.x, this.y);
        c.restore();
      }

      ctx.clearRect(0, 0, canvas.width, canvas.height);

        p.update();
        p.draw(ctx);
        // Clean up out of bounds particles or fully transparent ones

        return true;
      });

      animationFrameId = requestAnimationFrame(render);
    };

    return () => {
      window.removeEventListener("resize", resizeCanvas);
      window.removeEventListener("mousemove", handleMouseMove);
      window.removeEventListener("click", handleMouseClick);
      cancelAnimationFrame(animationFrameId);
    };
  }, [showHeartsRain, isAccepted]);

        ref={canvasRef}

        className="fixed inset-0 pointer-events-none z-20"
